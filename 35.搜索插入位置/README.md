class Solution {
    public int searchInsert(int[] nums, int target) {
        
        int start = 0;
        int end = nums.length-1;
        int mid=0;
        if(end==-1){
            return 0;
        }
        while(start <= end){
            mid = (start+end)/2;
            if(nums[mid]==target){
                return mid;
            }else if(nums[mid]<target){
                start = mid+1;
            }else if(nums[mid]>target){
                end = mid-1;
            }
        }
        
        return nums[mid] < target?mid+1:mid;
        
    }
}
