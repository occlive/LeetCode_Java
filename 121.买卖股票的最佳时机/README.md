    public int maxProfit(int[] prices) {

class Solution {
    public int maxProfit(int[] prices) {
        int max = 0;
        int minp = Integer.MAX_VALUE;
        for(int i = 0;i < prices.length;i++){
            if(prices[i] < minp){
                minp = prices[i];
            }else if(prices[i]-minp > max){
                max = prices[i]-minp;
            }
            
        }
        return max;
    }
}
