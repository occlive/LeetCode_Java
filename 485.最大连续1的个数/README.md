class Solution {
    public int findMaxConsecutiveOnes(int[] nums) {
        int num = 0,max = 0;
        for(int i:nums){
            if(i==1){
                num++;
            }else{
                max = num>max?num:max;
                num=0;
            }   
        }
        return max<num?num:max;
    }
}
