    public int[][] flipAndInvertImage(int[][] A) {
        int len = A[0].length-1;
        int[][] arr = new int[A.length][A[0].length];

                arr[i][len-j] = A[i][j];
            }
        }
        //åè½¬
        for(int i=0;i<A.length;i++){
            for(int j = 0;j<A[i].length;j++){

            }
        }
        return arr;
    }
}

class Solution {
    public int[][] flipAndInvertImage(int[][] A) {
        int C = A[0].length;
        for (int[] row: A)
            for (int i = 0; i < (C + 1) / 2; ++i) {
                int tmp = row[i] ^ 1;
                row[i] = row[C - 1 - i] ^ 1;
                row[C - 1 - i] = tmp;
            }

        return A;
    }
}
