    public boolean isPalindrome(int x) {
        if (x < 0 || x > 2147483647 || x < -2147483647) {
			return false;
		}
		StringBuffer sb = new StringBuffer(x + "");

		sb.reverse();

		long i = Long.parseLong(sb.toString());

		if (x == i) {
			return true;
		}

		return false;
    }
}

class Solution {
    public boolean isPalindrome(int x) {
        if ((x%10 == 0 && x != 0) || x < 0){
            return false;
        }
        int temp = x;
        int res = 0;
        while(x > 0){
            res = res * 10 + x % 10;
            x /= 10;
            
        }
        return temp == res;
    }
}
